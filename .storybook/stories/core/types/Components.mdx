# Component Types

Base props types shared between React and Vue component implementations. These types define the common interface for each component.

All types use the `T` prefix for easy identification in code.

## TButtonBaseProps

Base props shared between React and Vue Button components.

<Source
  language="ts"
  dark
  code={`import type { TButtonBaseProps } from "venomous-ui/core";

interface TButtonBaseProps {
/** Button size \*/
size?: TSize;
/** Button variant _/
variant?: TVariant;
/\*\* Color scheme _/
colorScheme?: TColorScheme;
/** Whether the button is disabled \*/
disabled?: boolean;
/** Whether the button takes full width _/
fullWidth?: boolean;
/\*\* Additional CSS class names _/
className?: string;
}`}
/>

<StoryDocsTable
  headers={["Prop", "Type", "Default", "Description"]}
  rows={[

    ["disabled", "boolean", "false", "Disables the button"],
    ["fullWidth", "boolean", "false", "Makes button full width"],
    ["className", "string", "-", "Additional CSS classes"],
  ]}
/>

## TTabsBaseProps

Base props for Tabs root component.

<Source
  language="ts"
  dark
  code={`import type { TTabsBaseProps, TTabsVariant, TOrientation, TTabsUnmountStrategy } from "venomous-ui/core";

type TOrientation = "horizontal" | "vertical";
type TTabsVariant = "line" | "pill";
type TTabsUnmountStrategy = "keepMounted" | "unmountOnHide";

interface TTabsBaseProps {
/** Default active tab value (uncontrolled mode) \*/
defaultValue?: string;
/** Current active tab value (controlled mode) \*/
value?: string;
/** Tabs orientation \*/
orientation?: TOrientation;
/** Visual variant \*/
variant?: TTabsVariant;
/\*\* Panel unmount strategy \*/
unmountStrategy?: TTabsUnmountStrategy;
}`}
/>
