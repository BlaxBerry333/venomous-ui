# useThemeMode

```tsx
import { useThemeMode } from 'venomous-ui';
```

## Examples

```tsx
import { ThemeProvider, useThemeMode, Button, Text, Flex } from 'venomous-ui';

function App() {
  return (
    <ThemeProvider>
      <Component />
    </ThemeProvider>
  );
}

function Component() {
  const { isDarkMode, themeMode, toggleThemeMode, setThemeMode } = useThemeMode();

  return (
    <Space>
      <Text text={themeMode} />
      <Button text={`change to ${isDarkMode ? 'Light' : 'Dark'}`} onClick={toggleThemeMode} />
      <Button
        text={`change to "Dark"`}
        disabled={isDarkMode}
        onClick={() => setThemeMode('dark')}
      />
      <Button
        text={`change to "Light"`}
        disabled={!isDarkMode}
        onClick={() => setThemeMode('light')}
      />
    </Space>
  );
}
```
