# useThemeBreakpoint

```tsx
import { useThemeBreakpoint } from 'venomous-ui';
```

## Examples

```tsx
import { useThemeBreakpoint } from 'venomous-ui';

function App() {
  const { isXs, isSm, isMd, isLg, isXl, breakpoints } = useThemeBreakpoint();
  const { values, current, unit } = breakpoints;

  return (
    <div>
      <p>current: {current}</p>
      <p>unit: {unit}</p>
      <br />
      <p>
        isXs: {isXs.toString()} {`${values['xs']}${unit}`}
      </p>
      <p>
        isSm: {isSm.toString()} {`${values['sm']}${unit}`}
      </p>
      <p>
        isMd: {isMd.toString()} {`${values['md']}${unit}`}
      </p>
      <p>
        isLg: {isLg.toString()} {`${values['lg']}${unit}`}
      </p>
      <p>
        isXl: {isLg.toString()} {`${values['xl']}${unit}`}
      </p>
    </div>
  );
}
```
