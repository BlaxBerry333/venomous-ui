# getColors

```tsx
import { getColors } from 'venomous-ui';
```

## Examples

```tsx
import { getColors } from 'venomous-ui';

const {
  origin, // #009688
  light, // rgb(127, 202, 195)
  dark, // rgb(0, 75, 68)
  opacity, // rgba(0, 150, 136, 0.2)
} = getColors('#009688');
```
