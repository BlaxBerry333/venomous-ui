# ThemeProvider

```tsx
import { ThemeProvider } from 'venomous-ui';
```

## Examples

#### ThemeMode

<Canvas
  of={Stories.ThemeMode}
  source={{
    code: [
      "import { ThemeProvider, useThemeMode, Button, Text, Flex } from 'venomous-ui';",
      '',
      'function Demo() {',
      '  const { isDarkMode, themeMode, toggleThemeMode, setThemeMode } = useThemeMode();',
      '',
      '  return (',
      '    <Flex>',
      '      <Text text={themeMode} />',
      '      <Button',
      '        text={`change to ${isDarkMode ? "Light" : "Dark"}`}',
      '        onClick={toggleThemeMode}',
      '      />',
      '      <Button',
      '        text={`change to "Dark"`}',
      '        disabled={isDarkMode}',
      '        onClick={() => setThemeMode("dark")}',
      '      />',
      '      <Button',
      '        text={`change to "Light"`}',
      '        disabled={!isDarkMode}',
      '        onClick={() => setThemeMode("light")}',
      '      />',
      '    </Flex>',
      '  );',
      '}',
      '',
      'function App() {',
      '  return (',
      '    <ThemeProvider>',
      '      <Demo />',
      '    </ThemeProvider>',
      '  );',
      '}',
    ].join('\n'),
  }}
/>
