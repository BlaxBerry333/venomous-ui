#### InsidePaper

It's highly recommended to add the `ellipsis` attribute when render a `<MenuItem/>` as the result of `renderItem` inside a fixed-size container.

<Canvas
  of={Stories.InsidePaper}
  sourceState="shown"
  source={{
    code: [
      "import { Menu, type MenuProps, MenuItem } from 'venomous-ui';",
      '',
      'function App() {',
      '  const menuItems: MenuProps["items"] = [',
      '    { label: "Item 1", icon: "solar:box-minimalistic-bold-duotone" },',
      '    { label: "Item 2", icon: "solar:box-minimalistic-bold-duotone" },',
      '  ];',
      '',
      '  return (',
      '    <Paper isOutlined sx={{ height: 200, width: 200 }}>',
      '      <Menu',
      '        height="100%"',
      '        width="100%"',
      '        items={menuItems}',
      '        renderItem={(item) => <MenuItem {...item} clickable ellipsis />}',
      '      />',
      '    </Paper>',
      '  );',
      '}',
    ].join('\n'),
  }}
/>

#### Virtualized

<Canvas
  of={Stories.Virtualized}
  sourceState="shown"

      '      width={300}',

      '      renderItem={(item, index) => (',
      '        <Flex row>',
      '          <Text text={`${index + 1}`} width={20}/>',
      '          <Text text={item.label} />',
      '        </Flex>',
      '      )}',
      '    />',
      '  );',
      '}',
    ].join('\n'),
  }}
/>
